HTML 에서 주석 쓰기는 다음과 같이 한다.

<!-- [주석] -- >

* 전체코드 상에서는 다음과 같이 한다(스크립트 내에서의 주석도 포함). 

          
            <!DOCTYPE html>
            <html>
              <head>
                <!-- 주석입니다 -->
                <script>
                  // 주석입니다
                  /*
                  주석입니다
                  */
                <script>
              </head>
              <body>
                <!-- <주석입니다> -->
              </body>
            </html>
